1//! Crate for creating sandbox environments to perform actions in isolation.
2//!
3//! This crate provides two main types:
4//! - [`Jailer`]: A simple sandbox that changes the current working directory
5//!   into a temporary one. When dropped or closed, it restores the original
6//!   directory and cleans up the temporary space.
7//! - [`EnvJailer`]: Extends [`Jailer`] by also managing environment variables,
8//!   allowing preservation of selected variables while clearing others on exit.
9//!
10//! Both types are thread-safe and ensure only one instance runs at a time via
11//! a global mutex.
12
13use std::collections::{HashMap, HashSet};
14use std::ffi::{OsStr, OsString};
15use std::path::PathBuf;
16use std::sync::{Arc, OnceLock};
17
18use parking_lot::lock_api::ArcMutexGuard;
19use parking_lot::{Mutex, RawMutex};
20use tempfile::TempDir;
21
22/// Static mutex which is wrapped around once lock to be thread safe
23static MUTEX: OnceLock<Arc<Mutex<()>>> = OnceLock::new();
24
25/// Initialize or get mutex
26fn initialize_or_get_mutex<'a>() -> &'a Arc<Mutex<()>> {
27    MUTEX.get_or_init(|| Arc::new(Mutex::new(())))
28}

30/// [`Jailer`] struct which creates a jail environment.
31///
32/// [`Jailer`] creates a temporary directory and changes the current working
33/// directory to it. On drop or manual close, it restores the original working
34/// directory and deletes the temporary directory.
35///
36/// It uses a global mutex to ensure only one `Jailer` is active at a time
37/// across threads.
38pub struct Jailer {
39    temp_directory: Option<TempDir>,
40    original_directory: PathBuf,
41    _lock: ArcMutexGuard<RawMutex, ()>,
42}
43
44impl Jailer {
45    /// Create a new [`Jailer`].
46    ///
47    /// This will:
48    /// - Lock globally to prevent concurrent instances.
49    /// - Create a temporary directory.
50    /// - Change the current directory to that temp dir.
51    ///
52    /// # Errors
53    ///
54    /// Returns an error if:
55    /// - The temporary directory cannot be created.
56    /// - Changing the current directory fails.
57    ///
58    /// # Example
59    ///
60    /// ```rust
61    /// use jailer::Jailer;
62    ///
63    /// // Capture the original working directory
64    /// let original_directory = std::env::current_dir().unwrap();
65    ///
66    /// let mut jailer = Jailer::new().unwrap();
67    ///
68    /// // We're now inside the jail
69    /// let inside_jailer_directory = std::env::current_dir().unwrap();
70    /// assert_ne!(inside_jailer_directory, original_directory);
71    ///
72    /// // Do some action in jailer...
73    ///
74    /// // Close the jail explicitly
75    /// jailer.close().unwrap();
76    ///
77    /// // Back to original directory
78    /// let after_jailer_directory = std::env::current_dir().unwrap();
79    /// assert_eq!(after_jailer_directory, original_directory);
80    /// assert_ne!(inside_jailer_directory, after_jailer_directory);
81    /// ```
82    pub fn new() -> Result<Self, std::io::Error> {
83        let lock = initialize_or_get_mutex().lock_arc();
84        let temp_dir = TempDir::new()?;
85        let original_directory = std::env::current_dir()?;
86        std::env::set_current_dir(&temp_dir)?;
87        Ok(Self {
88            temp_directory: Some(temp_dir),
89            original_directory,
90            _lock: lock,
91        })
92    }
93
94    /// Get a reference to the original directory
95    ///
96    /// This returns the directory that was active when the [`Jailer`] was
97    /// created
98    ///
99    /// # Example
100    ///
101    /// ```rust
102    /// use jailer::Jailer;
103    ///
104    /// let original_directory = std::env::current_dir().unwrap();
105    /// let jailer = Jailer::new().unwrap();
106    ///
107    /// assert_eq!(jailer.original_directory(), &original_directory);
108    /// jailer.close().unwrap();
109    /// ```
110    #[must_use]
111    pub fn original_directory(&self) -> &PathBuf {
112        &self.original_directory
113    }
114
115    /// Closes the [`Jailer`] and performs cleanup.
116    ///
117    /// This method:
118    /// - Changes back to the original working directory.
119    /// - Deletes the temporary directory.
120    /// - Releases the global lock.
121    ///
122    /// It consumes `self`, so the jailer cannot be used afterward.
123    ///
124    /// # Errors
125    ///
126    /// Returns an error if changing the directory or deleting the temp dir
127    /// fails.
128    pub fn close(mut self) -> Result<(), std::io::Error> {
129        std::env::set_current_dir(self.original_directory.as_path())?;
130        if let Some(temp) = self.temp_directory.take() {
131            temp.close()?;
132        }
133        std::mem::forget(self);
134        Ok(())
135    }
136}
137
138impl Drop for Jailer {
139    fn drop(&mut self) {
140        std::env::set_current_dir(self.original_directory.as_path()).ok();
141        if let Some(temp) = self.temp_directory.take() {
142            temp.close().ok();
143        }
144    }
145}

147/// [`EnvJailer`] struct which creates a jail environment with environment
148/// variable management.
149///
150/// [`EnvJailer`] wraps [`Jailer`] and adds support for preserving specific
151/// environment variables. On drop or close, it:
152/// - Removes all environment variables not marked as preserved.
153/// - Restores original values for preserved keys.
154/// - Reverts to the original working directory and cleans up the temp dir.
155///
156/// # Safety
157///
158/// Environment variable manipulation via [`std::env::set_var`] and
159/// [`std::env::remove_var`] is considered unsafe due to potential race
160/// conditions in multi-threaded programs. Therefore, methods like
161/// [`EnvJailer::close`] are marked as `unsafe`.
162pub struct EnvJailer {
163    jailer: Option<Jailer>,
164    original_directory: PathBuf,
165    original_env_vars_os: HashMap<OsString, OsString>,
166    preserved_env_vars_os: HashSet<OsString>,
167}
168
169impl EnvJailer {
170    /// Create a new [`EnvJailer`].
171    ///
172    /// This captures the current environment variables and working directory,
173    /// then initializes a new [`Jailer`].
174    ///
175    /// # Errors
176    ///
177    /// Returns an error if the underlying [`Jailer`] cannot be created.
178    ///
179    /// # Example
180    ///
181    /// ```rust
182    /// use jailer::EnvJailer;
183    ///
184    /// // Capture the original working directory
185    /// let original_directory = std::env::current_dir().unwrap();
186    ///
187    /// let mut env_jailer = EnvJailer::new().unwrap();
188    ///
189    /// // Do some action in jailer...
190    ///
191    /// // Close the jail explicitly (unsafe due to env var changes)
192    /// unsafe {
193    ///     env_jailer.close().unwrap();
194    /// }
195    ///
196    /// // Back to original directory
197    /// assert_eq!(std::env::current_dir().unwrap(), original_directory);
198    /// ```
199    pub fn new() -> Result<Self, std::io::Error> {
200        let original_env_vars_os = std::env::vars_os().collect();
201        let jailer = Jailer::new()?;
202        let original_dir = jailer.original_directory().clone();
203
204        Ok(Self {
205            jailer: Some(jailer),
206            original_directory: original_dir,
207            original_env_vars_os,
208            preserved_env_vars_os: HashSet::new(),
209        })
210    }
211
212    /// Get a reference to the original directory
213    ///
214    /// This returns the directory that was active when the [`EnvJailer`] was
215    /// created
216    ///
217    /// # Example
218    ///
219    /// ```rust
220    /// use jailer::EnvJailer;
221    ///
222    /// let original_directory = std::env::current_dir().unwrap();
223    /// let env_jailer = EnvJailer::new().unwrap();
224    /// assert_eq!(env_jailer.original_directory(), &original_directory);
225    /// unsafe {
226    ///     env_jailer.close().unwrap();
227    /// }
228    /// ```
229    #[must_use]
230    pub fn original_directory(&self) -> &PathBuf {
231        &self.original_directory
232    }
233
234    /// Mark an environment variable as preserved.
235    ///
236    /// When the jailer is closed or dropped, this key will retain its current
237    /// value instead of being removed or reset to the original value.
238    ///
239    /// # Example
240    ///
241    /// ```rust
242    /// use jailer::EnvJailer;
243    ///
244    /// unsafe {
245    ///     std::env::set_var("KEY", "VALUE");
246    ///     std::env::set_var("ANOTHER_KEY", "VALUE");
247    /// }
248    ///
249    /// let mut env_jailer = EnvJailer::new().unwrap();
250    ///
251    /// assert_eq!(std::env::var("KEY"), Ok("VALUE".to_string()));
252    ///
253    /// unsafe {
254    ///     std::env::set_var("KEY2", "VALUE2");
255    /// }
256    /// env_jailer.set_preserved_env("KEY");
257    ///
258    /// unsafe {
259    ///     std::env::set_var("KEY", "VALUE2");
260    ///     std::env::set_var("ANOTHER_KEY", "ANOTHER_VAL");
261    /// }
262    ///
263    /// assert_eq!(std::env::var("KEY"), Ok("VALUE2".to_string()));
264    /// assert_eq!(std::env::var("ANOTHER_KEY"), Ok("ANOTHER_VAL".to_string()));
265    ///
266    /// unsafe {
267    ///     env_jailer.close().unwrap();
268    /// }
269    ///
270    /// assert_eq!(std::env::var("KEY"), Ok("VALUE2".to_string()));
271    /// assert_eq!(std::env::var("ANOTHER_KEY"), Ok("VALUE".to_string()));
272    /// assert!(std::env::var("KEY2").is_err());
273    /// ```
274    pub fn set_preserved_env<K>(&mut self, key: K)
275    where
276        K: AsRef<OsStr>,
277    {
278        self.preserved_env_vars_os
279            .insert(key.as_ref().to_os_string());
280    }
281
282    /// Remove an environment variable from the preserved list.
283    ///
284    /// Note: This does *not* remove the current environment variable.
285    /// To remove it, call [`std::env::remove_var`] manually.
286    ///
287    /// # Example
288    ///
289    /// ```rust
290    /// use jailer::EnvJailer;
291    ///
292    /// let mut env_jailer = EnvJailer::new().unwrap();
293    ///
294    /// unsafe {
295    ///     std::env::set_var("KEY", "VALUE");
296    /// }
297    ///
298    /// env_jailer.set_preserved_env("KEY");
299    /// assert_eq!(std::env::var("KEY"), Ok("VALUE".to_string()));
300    /// env_jailer.remove_preserved_env("KEY");
301    ///
302    /// unsafe {
303    ///     env_jailer.close().unwrap();
304    /// }
305    ///
306    /// assert!(std::env::var("KEY").is_err());
307    /// ```
308    pub fn remove_preserved_env<K>(&mut self, key: K)
309    where
310        K: AsRef<OsStr>,
311    {
312        self.preserved_env_vars_os.remove(key.as_ref());
313    }
314
315    /// Returns a reference to the map of original environment variables.
316    ///
317    /// These are the environment variables present when the [`EnvJailer`] was
318    /// created. Any variables added during the session are not included
319    /// here.
320    #[must_use]
321    pub fn original_env_vars_os(&self) -> &HashMap<OsString, OsString> {
322        &self.original_env_vars_os
323    }
324
325    /// Returns a reference to the set of preserved environment variable names.
326    #[must_use]
327    pub fn preserved_env_vars_os(&self) -> &HashSet<OsString> {
328        &self.preserved_env_vars_os
329    }
330
331    unsafe fn revert_env_vars(&self) {
332        for key in std::env::vars_os().collect::<HashMap<_, _>>().keys() {
333            if !self.preserved_env_vars_os.contains(key) {
334                unsafe {
335                    std::env::remove_var(key);
336                }
337            }
338        }
339        for (key, value) in &self.original_env_vars_os {
340            if !self.preserved_env_vars_os.contains(key) {
341                unsafe {
342                    std::env::set_var(key, value);
343                }
344            }
345        }
346    }
347
348    /// Closes the [`EnvJailer`] and performs cleanup.
349    ///
350    /// This method:
351    /// - Reverts environment variables to their original state (except
352    ///   preserved ones).
353    /// - Closes the underlying [`Jailer`] (restoring directory and removing
354    ///   temp dir).
355    ///
356    /// It consumes `self`, so the jailer cannot be used afterward.
357    ///
358    /// # Errors
359    ///
360    /// Returns an error if the underlying [`Jailer::close`] fails.
361    ///
362    /// # Safety
363    ///
364    /// This function calls [`std::env::remove_var`] and [`std::env::set_var`],
365    /// which are unsafe due to possible data races in concurrent contexts.
366    pub unsafe fn close(mut self) -> Result<(), std::io::Error> {
367        unsafe {
368            self.revert_env_vars();
369        }
370        if let Some(jailer) = self.jailer.take() {
371            jailer.close()?;
372        }
373        std::mem::forget(self);
374        Ok(())
375    }
376}
377
378impl Drop for EnvJailer {
379    fn drop(&mut self) {
380        unsafe {
381            self.revert_env_vars();
382        }
383    }
